# Dia 184 — Kotlin: Coroutines

## Conceito do Dia

**Coroutines** são threads leves para programação assíncrona não-bloqueante.

---

## Comparação

```kotlin
// Blocking (tradicional)
fun buscarDados(): List<Dado> {
    return dao.buscarTodos() // bloqueia thread
}

// Non-blocking (coroutines)
suspend fun buscarDados(): List<Dado> {
    return withContext(Dispatchers.IO) {
        dao.buscarTodos()
    }
}
```

## Exemplo com Service

```kotlin
@Service
class TarefaService(
    private val repository: TarefaRepository
) {
    suspend fun criar(request: CriarTarefaRequest): Tarefa {
        return withContext(Dispatchers.IO) {
            val tarefa = Tarefa(
                nome = request.nome,
                categoria = request.categoria,
                prioridade = request.prioridade
            )
            repository.save(tarefa)
        }
    }
    
    suspend fun buscarPorId(id: Long): Tarefa? {
        return withContext(Dispatchers.IO) {
            repository.findById(id).orElse(null)
        }
    }
}
```

## Controller Assíncrono

```kotlin
@RestController
@RequestMapping("/api/tarefas")
class TarefaController(
    private val service: TarefaService
) {
    @GetMapping("/{id}")
    suspend fun buscarPorId(@PathVariable id: Long) = 
        service.buscarPorId(id)
            ?: throw RecursoNaoEncontradoException("Tarefa", id)
    
    @GetMapping
    suspend fun listar(): List<Tarefa> = service.listarTodas()
}
```

## Parallel Execution

```kotlin
suspend fun buscarDashboard(): Dashboard {
    // Execução paralela
    val usuario = async { usuarioService.buscar() }
    val tarefas = async { tarefaService.listar() }
    val estatisticas = async { estatisticaService.calcular() }
    
    // Aguarda todos completarem
    return Dashboard(
        usuario = usuario.await(),
        tarefas = tarefas.await(),
        estatisticas = estatisticas.await()
    )
}
```

@Repository
interface TarefaRepository : JpaRepository<Tarefa, Long> {
    
    @Query("SELECT t FROM Tarefa t WHERE t.categoria = :cat")
