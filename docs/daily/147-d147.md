## Simple Memory (na sessão)

```java
@Service
@RequiredArgsConstructor
public class ChatService {
    
    private final ChatClient chatClient;
    private final Map<String, List<ChatMessage>> sessions = new ConcurrentHashMap<>();
    
    public String chat(String sessionId, String userMessage) {
        List<ChatMessage> history = sessions.computeIfAbsent(sessionId, k -> new ArrayList<>());
        
        // Adicionar mensagem do usuário
        history.add(new ChatMessage(MessageType.USER, userMessage));
        
        // Manter só últimas 10 mensagens
        if (history.size() > 20) {
            history = history.subList(history.size() - 20, history.size());
            sessions.put(sessionId, history);
        }
        
        // Construir prompt com histórico
        String prompt = buildPromptWithHistory(history);
        
        String response = chatClient.call(prompt);
        
        // Salvar resposta
        history.add(new ChatMessage(MessageType.ASSISTANT, response));
        
        return response;
    }
    
    private String buildPromptWithHistory(List<ChatMessage> history) {
        StringBuilder sb = new StringBuilder();
        sb.append("Histórico da conversa:\n");
        
        for (ChatMessage msg : history) {
            sb.append(msg.type() == MessageType.USER ? "Usuário: " : "Assistente: ");
            sb.append(msg.content()).append("\n");
        }
        
        return sb.toString();
    }
}
```

## ChatHistory com Spring AI

```java
@Service
@RequiredArgsConstructor
public class AwareChatService {
    
    private final ChatClient chatClient;
    private final VectorStore vectorStore;
    
    public String chat(ChatRequest request) {
        // 1. Buscar conversas similares no vector store
        List<Document> similarChats = vectorStore.similaritySearch(
            SearchRequest.builder()
                .query(request.message())
                .topK(3)
                .build()
        );
        
        // 2. Construir contexto
        String contextoAnterior = similarChats.stream()
            .map(Document::getText)
            .collect(Collectors.joining("\n\n"));
        
        String prompt = """
            Contexto de conversas anteriores:
            %s
            
            Conversa atual:
            Usuário: %s
            """.formatted(contextoAnterior, request.message());
        
        // 3. Responder
        String resposta = chatClient.call(prompt);
        
        // 4. Salvar esta conversa
        Document chatDoc = Document.builder()
            .text("Usuário: " + request.message() + "\nAssistente: " + resposta)
            .metadata(Map.of("sessionId", request.sessionId()))
            .build();
        
        vectorStore.write(List.of(chatDoc));
        
        return resposta;
    }
}
```

## Chat WebSocket

```java
@Configuration
@EnableWebSocket
public class WebSocketConfig implements WebSocketConfigurer {
    
    @Override
    public void registerWebSocketHandlers(WebSocketHandlerRegistry registry) {
        registry.addHandler(chatHandler(), "/ws/chat");
    }
}

@Component
@RequiredArgsConstructor
public class ChatWebSocketHandler extends TextWebSocketHandler {
    
    private final ChatService chatService;
    private final Map<String, WebSocketSession> sessions = new ConcurrentHashMap<>();
    
    @Override
    public void afterConnectionEstablished(WebSocketSession session) {
        sessions.put(session.getId(), session);
    }
    
    @Override
    protected void handleTextMessage(WebSocketSession session, TextMessage message) throws Exception {
        ChatRequest request = new ObjectMapper().readValue(message.getPayload(), ChatRequest.class);
        
        String response = chatService.chat(session.getId(), request.message());
        
        session.sendMessage(new TextMessage(
            new ObjectMapper().writeValueAsString(new ChatResponse(response))
        ));
    }
}
```
