## Rotas

```jsx
// App.jsx
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom';
import { useAuth } from './context/AuthContext';

function ProtectedRoute({ children }) {
    const { isAuthenticated } = useAuth();
    return isAuthenticated ? children : <Navigate to="/login" />;
}

function App() {
    return (
        <BrowserRouter>
            <Routes>
                <Route path="/login" element={<LoginPage />} />
                <Route path="/register" element={<RegisterPage />} />
                
                <Route path="/" element={
                    <ProtectedRoute>
                        <Layout />
                    </ProtectedRoute>
                }>
                    <Route index element={<Dashboard />} />
                    <Route path="tarefas" element={<TarefasPage />} />
                    <Route path="tarefas/:id" element={<TarefaDetail />} />
                    <Route path="perfil" element={<PerfilPage />} />
                </Route>
                
                <Route path="*" element={<Navigate to="/" />} />
            </Routes>
        </BrowserRouter>
    );
}
```

## useNavigate

```jsx
import { useNavigate } from 'react-router-dom';

function TarefasPage() {
    const navigate = useNavigate();
    
    const handleCriar = () => {
        navigate('/tarefas/nova');
    };
    
    return (
        <div>
            <button onClick={handleCriar}>Nova Tarefa</button>
        </div>
    );
}
```

## useParams

```jsx
function TarefaDetail() {
    const { id } = useParams();
    
    const { data: tarefa, isLoading } = useQuery({
        queryKey: ['tarefa', id],
        queryFn: () => api.buscarTarefa(id)
    });
    
    return (
        <div>
            <h2>Tarefa #{id}</h2>
            {isLoading ? <p>Carregando...</p> : <p>{tarefa.nome}</p>}
        </div>
    );
}
```
