## Componentes

```jsx
// TarefaCard.jsx
function TarefaCard({ tarefa, onConcluir, onDeletar }) {
    return (
        <div className={`tarefa-card ${tarefa.concluida ? 'concluida' : ''}`}>
            <div className="tarefa-info">
                <h3>{tarefa.nome}</h3>
                <span className="categoria">{tarefa.categoria}</span>
            </div>
            <div className="tarefa-actions">
                <button onClick={() => onConcluir(tarefa.id)}>
                    {tarefa.concluida ? '↩️' : '✅'}
                </button>
                <button onClick={() => onDeletar(tarefa.id)}>🗑️</button>
            </div>
        </div>
    );
}

// Uso
function App() {
    const tarefas = [{ id: 1, nome: 'Estudar React', categoria: 'ESTUDO', concluida: false }];
    
    return (
        <div>
            {tarefas.map(t => (
                <TarefaCard 
                    key={t.id}
                    tarefa={t}
                    onConcluir={handleConcluir}
                    onDeletar={handleDeletar}
                />
            ))}
        </div>
    );
}
```

## Composição

```jsx
// Card.jsx (genérico)
function Card({ children, className }) {
    return <div className={`card ${className || ''}`}>{children}</div>;
}

// Header.jsx
function Header({ titulo, subtitulo }) {
    return (
        <header>
            <h1>{titulo}</h1>
            {subtitulo && <p>{subtitulo}</p>}
        </header>
    );
}

// Uso
<Card>
    <Header titulo="Stay Hard" subtitulo="Suas tarefas" />
    <ListaTarefas tarefas={tarefas} />
</Card>
```
