## Criar Customer e Subscription

```java
@Service
@RequiredArgsConstructor
public class StripeService {
    
    private final Stripe stripe;
    
    public Customer criarCustomer(String email, String name) {
        Map<String, Object> params = Map.of(
            "email", email,
            "name", name
        );
        return stripe.customers().create(params);
    }
    
    public String criarCheckoutSession(String customerId, String priceId) {
        Map<String, Object> params = Map.of(
            "customer", customerId,
            "mode", "subscription",
            "success_url", "https://app.stayhard.com/success",
            "cancel_url", "https://app.stayhard.com/pricing",
            "line_items", List.of(Map.of(
                "price", priceId,
                "quantity", 1
            ))
        );
        
        Session session = stripe.checkout().sessions().create(params);
        return session.getUrl();
    }
    
    public void cancelSubscription(String subscriptionId) {
        stripe.subscriptions().cancel(subscriptionId, null);
    }
}
```

## Webhook Handler

```java
@RestController
@RequestMapping("/api/webhooks/stripe")
@RequiredArgsConstructor
public class StripeWebhookController {
    
    private final Stripe stripe;
    private final AssinaturaService assinaturaService;
    
    @PostMapping
    public ResponseEntity<Void> handleWebhook(
            @RequestBody String payload,
            @RequestHeader("Stripe-Signature") String sig) {
        
        Event event = Webhook.constructEvent(payload, sig, 
            stripeProperties.getWebhookSecret());
        
        switch (event.getType()) {
            case "customer.subscription.created":
            case "customer.subscription.updated":
                handleSubscriptionChange(event);
                break;
            case "customer.subscription.deleted":
                handleSubscriptionCancelled(event);
                break;
            case "invoice.payment_failed":
                handlePaymentFailed(event);
                break;
        }
        
        return ResponseEntity.ok().build();
    }
}
```
