## 💻 CHECKLIST DE CÓDIGO

### 📁 Exercício 1: INSERT

```sql
-- Inserir dados
INSERT INTO usuarios (nome, email, idade)
VALUES ('André', 'andre@email.com', 25);

-- Inserir múltiplos
INSERT INTO usuarios (nome, email, idade)
VALUES 
    ('Maria', 'maria@email.com', 30),
    ('João', 'joao@email.com', 28),
    ('Ana', 'ana@email.com', 22);

-- Inserir com defaults
INSERT INTO usuarios (nome, email) VALUES ('Pedro', 'pedro@email.com');
-- ativo = TRUE (default), data_criacao = NOW() (default)

-- Inserir sem especificar colunas (todas)
INSERT INTO usuarios VALUES (1, 'Teste', 'test@test.com', 20, TRUE, NOW());
```

### 📁 Exercício 2: SELECT Básico

```sql
-- Selecionar tudo
SELECT * FROM usuarios;

-- Selecionar colunas específicas
SELECT nome, email FROM usuarios;

-- Selecionar com alias
SELECT nome AS 'Nome do Usuário', email AS Email FROM usuarios;

-- Contar registros
SELECT COUNT(*) FROM usuarios;

-- Selecionar sem duplicatas
SELECT DISTINCT cidade FROM usuarios;
```

---

### 📁 Exercício 3: WHERE

```sql
-- Filtrar por condição
SELECT * FROM usuarios WHERE idade >= 18;

-- Múltiplas condições
SELECT * FROM usuarios 
WHERE idade >= 18 AND ativo = TRUE;

-- OR
SELECT * FROM usuarios 
WHERE idade < 18 OR cidade = 'São Paulo';

-- LIKE (busca parcial)
SELECT * FROM usuarios WHERE nome LIKE 'A%';       -- Começa com A
SELECT * FROM usuarios WHERE email LIKE '%@gmail.com'; -- Termina com gmail

-- IN (lista de valores)
SELECT * FROM usuarios WHERE cidade IN ('São Paulo', 'Rio', 'BH');

-- BETWEEN (range)
SELECT * FROM usuarios WHERE idade BETWEEN 18 AND 30;

-- IS NULL / IS NOT NULL
SELECT * FROM usuarios WHERE telefone IS NULL;
SELECT * FROM usuarios WHERE telefone IS NOT NULL;
```

---

### 📁 Exercício 4: ORDER BY e LIMIT

```sql
-- Ordenar
SELECT * FROM usuarios ORDER BY nome;              -- Alfabético
SELECT * FROM usuarios ORDER BY nome DESC;         -- Invertido
SELECT * FROM usuarios ORDER BY idade, nome;       -- Múltiplos

-- LIMIT
SELECT * FROM usuarios LIMIT 10;                  -- Primeiros 10
SELECT * FROM usuarios LIMIT 5 OFFSET 10;           -- 5 a partir do 10
SELECT * FROM usuarios ORDER BY id DESC LIMIT 1;   -- Último registro
```

---

### 📁 Exercício 5: Funções Agregadas

```sql
-- COUNT
SELECT COUNT(*) FROM usuarios;
SELECT COUNT(*) FROM usuarios WHERE ativo = TRUE;

-- SUM
SELECT SUM(idade) FROM usuarios;
SELECT SUM(salario) FROM usuarios WHERE cargo = 'Dev';

-- AVG (média)
SELECT AVG(idade) FROM usuarios;

-- MIN / MAX
SELECT MIN(idade), MAX(idade) FROM usuarios;

-- GROUP BY
SELECT cidade, COUNT(*) as total 
FROM usuarios 
GROUP BY cidade;
```
