## Arquivos principais

```javascript
// store.js
class Store {
    constructor() {
        this.state = {
            tarefas: [],
            filtro: 'TODAS',
            loading: false,
            erro: null,
            token: localStorage.getItem('token'),
            usuario: JSON.parse(localStorage.getItem('usuario') || 'null')
        };
        this.listeners = new Set();
    }
    
    update(partial) {
        this.state = { ...this.state, ...partial };
        this.notify();
    }
    
    subscribe(fn) {
        this.listeners.add(fn);
        return () => this.listeners.delete(fn);
    }
    
    notify() {
        this.listeners.forEach(fn => fn(this.state));
    }
}

export const store = new Store();
```

```javascript
// components.js
export function renderApp(state) {
    if (!state.token) {
        return renderAuthScreen(state);
    }
    return renderMainScreen(state);
}

export function renderMainScreen(state) {
    return `
        <div class="app">
            <header>
                <h1>🔥 Stay Hard</h1>
                <div class="user-info">
                    <span>${state.usuario?.nome}</span>
                    <button onclick="auth.logout()">Sair</button>
                </div>
            </header>
            <main>
                ${renderFormulario()}
                ${renderFiltros(state.filtro)}
                ${renderListaTarefas(state.tarefas)}
            </main>
        </div>
    `;
}
```
