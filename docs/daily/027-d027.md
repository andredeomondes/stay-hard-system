## 💻 CHECKLIST DE CÓDIGO

### 📁 Exercício 1: UPDATE

```sql
-- Atualizar TODOS (perigoso!)
UPDATE usuarios SET ativo = FALSE;

-- Atualizar com WHERE
UPDATE usuarios SET ativo = FALSE WHERE id = 1;

-- Atualizar múltiplas colunas
UPDATE usuarios 
SET idade = 26, cidade = 'São Paulo' 
WHERE email = 'andre@email.com';

-- Atualizar baseado em condição
UPDATE usuarios 
SET status = 'premium' 
WHERE data_criacao < '2024-01-01';

-- UPDATE com CASE (atualização condicional)
UPDATE produtos
SET preco = CASE
    WHEN categoria = 'eletronicos' THEN preco * 0.9
    WHEN categoria = 'roupas' THEN preco * 0.8
    ELSE preco
END;
```

---

### 📁 Exercício 2: DELETE

```sql
-- Deletar específico
DELETE FROM usuarios WHERE id = 5;

-- Deletar múltiplos
DELETE FROM usuarios WHERE cidade = 'Rio de Janeiro';

-- Deletar com subquery
DELETE FROM usuarios 
WHERE id IN (SELECT id FROM usuarios_inativos WHERE meses_sem_login > 12);

-- Deletar TODOS (perigoso!)
DELETE FROM usuarios;

-- Deletar rápido (sem log)
TRUNCATE TABLE usuarios;
```

---

### 📁 Exercício 3: WHERE Avançado

```sql
-- EXISTS (verifica se existe)
SELECT * FROM usuarios u
WHERE EXISTS (
    SELECT 1 FROM pedidos p WHERE p.usuario_id = u.id
);

-- NOT EXISTS
SELECT * FROM usuarios u
WHERE NOT EXISTS (
    SELECT 1 FROM pedidos p WHERE p.usuario_id = u.id
);

-- ALL (todos devem satisfazer)
SELECT * FROM usuarios 
WHERE idade > ALL (SELECT idade FROM usuarios WHERE cidade = 'SP');

-- ANY (pelo menos um satisfaz)
SELECT * FROM produtos 
WHERE preco > ANY (SELECT preco FROM produtos_em_promocao);
```
