## Índices

```sql
-- Índice B-tree (padrão)
CREATE INDEX idx_tarefa_categoria ON tarefas(categoria);

-- Índice composto
CREATE INDEX idx_tarefa_categoria_concluida ON tarefas(categoria, concluida);

-- Índice parcial (só pendentes)
CREATE INDEX idx_tarefa_pendentes ON tarefas(criado_em) 
WHERE concluida = false;

-- Índice para busca de texto
CREATE INDEX idx_tarefa_nome_trgm ON tarefas USING gin(nome gin_trgm_ops);
```

    Projeto p = projetoRepo.findById(t.getProjetoId()); // 1 query por tarefa!

## Batch Inserts

```java
@Service
@RequiredArgsConstructor
public class TarefaService {
    
    @PersistenceContext
    private EntityManager em;
    
    public void salvarEmMassa(List<TarefaDTO> tarefas) {
        int batchSize = 50;
        int i = 0;
        
        for (TarefaDTO dto : tarefas) {
            Tarefa tarefa = mapear(dto);
            em.persist(tarefa);
            i++;
            
            if (i % batchSize == 0) {
                em.flush();
                em.clear();
            }
        }
        em.flush();
        em.clear();
    }
}
```
