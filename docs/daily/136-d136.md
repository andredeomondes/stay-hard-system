## Docker Compose

```yaml
kafka:
  image: confluentinc/cp-kafka:7.6.0
  ports:
    - "9092:9092"
  environment:
    KAFKA_BROKER_ID: 1
    KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
    KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://kafka:29092,PLAINTEXT_HOST://localhost:9092
    KAFKA_LISTENER_SECURITY_PROTOCOL_MAP: PLAINTEXT:PLAINTEXT,PLAINTEXT_HOST:PLAINTEXT
    KAFKA_INTER_BROKER_LISTENER_NAME: PLAINTEXT
    KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
```

## Producer

```java
@Service
@RequiredArgsConstructor
public class TarefaEventProducer {
    
    private final KafkaTemplate<String, TarefaEvent> kafkaTemplate;
    
    public void enviarTarefaCriada(Tarefa tarefa) {
        TarefaEvent event = new TarefaEvent(
            "TAREFA_CRIADA",
            tarefa.getId(),
            tarefa.getNome(),
            LocalDateTime.now()
        );
        
        kafkaTemplate.send("tarefas-events", tarefa.getId().toString(), event)
            .whenComplete((result, ex) -> {
                if (ex == null) {
                    log.info("Evento enviado: {}", result.getRecordMetadata());
                } else {
                    log.error("Erro ao enviar evento", ex);
                }
            });
    }
}
```

## Consumer

```java
@Service
@RequiredArgsConstructor
public class NotificacaoConsumer {
    
    @KafkaListener(topics = "tarefas-events", groupId = "notificacao-group")
    public void consumirTarefaCriada(TarefaEvent event) {
        if ("TAREFA_CRIADA".equals(event.tipo())) {
            log.info("Enviando notificação para tarefa: {}", event.nome());
            emailService.enviar(event.nome());
        }
    }
}
```
