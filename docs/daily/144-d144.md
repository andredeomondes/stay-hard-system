## Spring AI Embeddings

```java
@Service
@RequiredArgsConstructor
public class EmbeddingService {
    
    private final EmbeddingModel embeddingModel;
    
    public float[] gerarEmbedding(String texto) {
        EmbeddingRequest request = new EmbeddingRequest(
            List.of(texto),
            OpenAiEmbeddingOptions.builder()
                .model("text-embedding-3-small")
                .build()
        );
        
        EmbeddingResponse response = embeddingModel.call(request);
        return response.getResult().getEmbedding();
    }
    
    public double similaridade(String texto1, String texto2) {
        float[] emb1 = gerarEmbedding(texto1);
        float[] emb2 = gerarEmbedding(texto2);
        
        return cossenoSimilarity(emb1, emb2);
    }
    
    private double cossenoSimilarity(float[] a, float[] b) {
        double dotProduct = 0;
        double normA = 0;
        double normB = 0;
        
        for (int i = 0; i < a.length; i++) {
            dotProduct += a[i] * b[i];
            normA += a[i] * a[i];
            normB += b[i] * b[i];
        }
        
        return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
    }
}
```

## Uso para Similaridade de Tarefas

```java
@Service
@RequiredArgsConstructor
public class TarefaSimilaridadeService {
    
    private final EmbeddingService embeddingService;
    private final TarefaRepository repository;
    
    public List<Tarefa> encontrarTarefasSimilares(Tarefa tarefa) {
        float[] embedding = embeddingService.gerarEmbedding(tarefa.getNome());
        
        List<Tarefa> tarefas = repository.findAll();
        
        return tarefas.stream()
            .filter(t -> !t.getId().equals(tarefa.getId()))
            .map(t -> {
                float[] emb = embeddingService.gerarEmbedding(t.getNome());
                double similarity = cosineSimilarity(embedding, emb);
                return new AbstractMap.SimpleEntry<>(t, similarity);
            })
            .filter(e -> e.getValue() > 0.8)
            .sorted((a, b) -> Double.compare(b.getValue(), a.getValue()))
            .map(Map.Entry::getKey)
            .toList();
    }
}
```
