## Pub/Sub

```java
@Service
public class TarefaNotifier {
    
    private final RedisTemplate<String, Object> redisTemplate;
    
    public void notificarTarefaCriada(Tarefa tarefa) {
        redisTemplate.convertAndSend("tarefas:created", tarefa);
    }
    
    public void subscribe(Consumer<Tarefa> consumer) {
        redisTemplate.listenToChannel("tarefas:created", msg -> {
            Tarefa tarefa = (Tarefa) msg.getMessageObject();
            consumer.accept(tarefa);
        });
    }
}
```

## Rate Limiting

```java
@Service
public class RateLimitingService {
    
    private final RedisTemplate<String, String> redisTemplate;
    
    public boolean isAllowed(String userId) {
        String key = "rate:" + userId;
        
        Long current = redisTemplate.opsForValue().increment(key);
        
        if (current == 1) {
            redisTemplate.expire(key, Duration.ofMinutes(1));
        }
        
        return current <= 100; // 100 requests/min
    }
}
```

// Uso
@RestController
public class SessionController {
    
    @GetMapping("/sessao")
    public Map<String, Object> getSession(HttpSession session) {
        Map<String, Object> info = new HashMap<>();
        info.put("id", session.getId());
        info.put("criadoEm", session.getCreationTime());
        info.put("usuario", session.getAttribute("usuario"));
        return info;
    }
}

## Distributed Lock

```java
@Service
public class DistributedLockService {
    
    private final RedissonClient redisson;
    
    public <T> T executeWithLock(String key, int waitTime, 
                                Supplier<T> action) {
        RLock lock = redisson.getLock(key);
        
        try {
            if (lock.tryLock(waitTime, TimeUnit.SECONDS)) {
                return action.get();
            }
            throw new LockException("Não foi possível adquirir lock: " + key);
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
}
```
