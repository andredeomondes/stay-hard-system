### 2. Database.java — Conexão

```java
package com.estudo;

import java.sql.*;

public class Database {
    private static final String URL = "jdbc:sqlite:estoque.db";
    
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL);
    }
    
    public static void initDatabase() {
        String sql = """
            CREATE TABLE IF NOT EXISTS produtos (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                nome TEXT NOT NULL,
                preco REAL NOT NULL,
                quantidade INTEGER DEFAULT 0
            )
            """;
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement()) {
            stmt.execute(sql);
            System.out.println("Banco inicializado!");
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
}
```

### 4. ProdutoRepository.java — CRUD

```java
package com.estudo;

import java.sql.*;
import java.util.*;

public class ProdutoRepository {
    
    // CREATE
    public void salvar(Produto p) {
        String sql = "INSERT INTO produtos (nome, preco, quantidade) VALUES (?, ?, ?)";
        try (Connection conn = Database.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setString(1, p.getNome());
            stmt.setDouble(2, p.getPreco());
            stmt.setInt(3, p.getQuantidade());
            stmt.executeUpdate();
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
    
    // READ ALL
    public List<Produto> buscarTodos() {
        String sql = "SELECT * FROM produtos";
        List<Produto> produtos = new ArrayList<>();
        try (Connection conn = Database.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {
            while (rs.next()) {
                produtos.add(mapear(rs));
            }
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
        return produtos;
    }
    
    // READ ONE
    public Optional<Produto> buscarPorId(Long id) {
        String sql = "SELECT * FROM produtos WHERE id = ?";
        try (Connection conn = Database.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setLong(1, id);
            ResultSet rs = stmt.executeQuery();
            if (rs.next()) {
                return Optional.of(mapear(rs));
            }
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
        return Optional.empty();
    }
    
    // UPDATE
    public void atualizar(Produto p) {
        String sql = "UPDATE produtos SET nome = ?, preco = ?, quantidade = ? WHERE id = ?";
        try (Connection conn = Database.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setString(1, p.getNome());
            stmt.setDouble(2, p.getPreco());
            stmt.setInt(3, p.getQuantidade());
            stmt.setLong(4, p.getId());
            stmt.executeUpdate();
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
    
    // DELETE
    public void deletar(Long id) {
        String sql = "DELETE FROM produtos WHERE id = ?";
        try (Connection conn = Database.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setLong(1, id);
            stmt.executeUpdate();
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
    
    private Produto mapear(ResultSet rs) throws SQLException {
        Produto p = new Produto();
        p.setId(rs.getLong("id"));
        p.setNome(rs.getString("nome"));
        p.setPreco(rs.getDouble("preco"));
        p.setQuantidade(rs.getInt("quantidade"));
        return p;
    }
}
```

### 5. App.java — Main

```java
package com.estudo;

public class App {
    public static void main(String[] args) {
        Database.initDatabase();
        ProdutoRepository repo = new ProdutoRepository();
        
        // CREATE
        repo.salvar(new Produto("Teclado", 150.0, 10));
        repo.salvar(new Produto("Mouse", 80.0, 25));
        
        // READ
        System.out.println("Todos: " + repo.buscarTodos());
        System.out.println("Por ID: " + repo.buscarPorId(1L));
        
        // UPDATE
        Produto p = repo.buscarPorId(1L).get();
        p.setPreco(120.0);
        repo.atualizar(p);
        
        // DELETE
        repo.deletar(2L);
    }
}
```
