### 2. Calculadora.java

- [ ] Classe com operações matemáticas
- [ ] Tratamento de divisão por zero
- [ ] Testes unitários

```java
package com.estudo;

public class Calculadora {
    
    public int somar(int a, int b) {
        return a + b;
    }
    
    public int subtrair(int a, int b) {
        return a - b;
    }
    
    public int multiplicar(int a, int b) {
        return a * b;
    }
    
    public double dividir(int a, int b) {
        if (b == 0) {
            throw new ArithmeticException("Divisão por zero não é permitida");
        }
        return (double) a / b;
    }
    
    public double potencia(int base, int expoente) {
        return Math.pow(base, expoente);
    }
    
    public int fatorial(int n) {
        if (n < 0) {
            throw new IllegalArgumentException("Fatorial não existe para números negativos");
        }
        if (n == 0 || n == 1) {
            return 1;
        }
        return n * fatorial(n - 1);
    }
}
```

### 3. CalculadoraTest.java

- [ ] Testes para todas operações
- [ ] Testes de exception
- [ ] Todos passando

```java
package com.estudo;

import org.junit.jupiter.api.*;
import static org.junit.jupiter.api.Assertions.*;

class CalculadoraTest {
    
    private Calculadora calc;
    
    @BeforeEach
    void setup() {
        calc = new Calculadora();
    }
    
    // SOMA
    @Test
    @DisplayName("Soma de dois números positivos")
    void testSomaPositivos() {
        assertEquals(5, calc.somar(2, 3));
    }
    
    @Test
    @DisplayName("Soma com número negativo")
    void testSomaNegativo() {
        assertEquals(0, calc.somar(-2, 2));
    }
    
    @Test
    @DisplayName("Soma com zeros")
    void testSomaZeros() {
        assertEquals(0, calc.somar(0, 0));
    }
    
    // SUBTRAÇÃO
    @Test
    @DisplayName("Subtração normal")
    void testSubtrair() {
        assertEquals(5, calc.subtrair(10, 5));
    }
    
    // MULTIPLICAÇÃO
    @Test
    @DisplayName("Multiplicação")
    void testMultiplicar() {
        assertEquals(12, calc.multiplicar(3, 4));
    }
    
    // DIVISÃO
    @Test
    @DisplayName("Divisão normal")
    void testDividir() {
        assertEquals(2.0, calc.dividir(10, 5));
    }
    
    @Test
    @DisplayName("Divisão com decimal")
    void testDividirDecimal() {
        assertEquals(2.5, calc.dividir(5, 2));
    }
    
    @Test
    @DisplayName("Divisão por zero lança exceção")
    void testDividirPorZero() {
        assertThrows(ArithmeticException.class, () -> calc.dividir(10, 0));
    }
    
    // POTÊNCIA
    @Test
    @DisplayName("Potência")
    void testPotencia() {
        assertEquals(8, calc.potencia(2, 3));
    }
    
    // FATORIAL
    @Test
    @DisplayName("Fatorial de 5")
    void testFatorial() {
        assertEquals(120, calc.fatorial(5));
    }
    
    @Test
    @DisplayName("Fatorial de 0")
    void testFatorialZero() {
        assertEquals(1, calc.fatorial(0));
    }
    
    @Test
    @DisplayName("Fatorial de número negativo lança exceção")
    void testFatorialNegativo() {
        assertThrows(IllegalArgumentException.class, () -> calc.fatorial(-5));
    }
}
```
