## app.js

```javascript
const API = new StayHardAPI('http://localhost:8080');

async function init() {
    await carregarTarefas();
    setupEventListeners();
}

async function carregarTarefas() {
    const tarefas = await API.listarTarefas();
    renderizarTarefas(tarefas);
    atualizarContador(tarefas.length);
}

function renderizarTarefas(tarefas) {
    const container = document.getElementById('lista-tarefas');
    
    if (tarefas.length === 0) {
        container.innerHTML = '<p class="empty-state">Nenhuma tarefa ainda. Vamos começar! 🚀</p>';
        return;
    }
    
    container.innerHTML = tarefas.map(tarefa => criarCardTarefa(tarefa)).join('');
    
    // Adicionar event listeners aos botões
    container.querySelectorAll('.btn-concluir').forEach(btn => {
        btn.addEventListener('click', () => concluirTarefa(btn.dataset.id));
    });
}

function setupEventListeners() {
    document.getElementById('form-tarefa').addEventListener('submit', async (e) => {
        e.preventDefault();
        
        const dados = {
            nome: document.getElementById('tarefa-nome').value,
            categoria: document.getElementById('tarefa-categoria').value,
            prioridade: document.getElementById('tarefa-prioridade').value
        };
        
        try {
            await API.criarTarefa(dados);
            document.getElementById('form-tarefa').reset();
            await carregarTarefas();
        } catch (error) {
            alert('Erro ao criar tarefa');
        }
    });
}

document.addEventListener('DOMContentLoaded', init);
```
