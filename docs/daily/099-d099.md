## Testes com @WithMockUser

```java
@WebMvcTest(TarefaController.class)
@AutoConfigureMockMvc
class TarefaControllerTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @MockBean
    private TarefaService service;
    
    @Test
    void listar_comUsuarioAutenticado_deveRetornar200() throws Exception {
        when(service.listarTodas()).thenReturn(List.of());
        
        mockMvc.perform(get("/api/tarefas")
                .with(user("user@test.com").roles("USER")))
            .andExpect(status().isOk());
    }
    
    @Test
    @WithMockUser(roles = "ADMIN")
    void deletar_comAdmin_deveRetornar204() throws Exception {
        mockMvc.perform(delete("/api/tarefas/1"))
            .andExpect(status().isNoContent());
    }
    
    @Test
    void deletar_semAutenticacao_deveRetornar401() throws Exception {
        mockMvc.perform(delete("/api/tarefas/1"))
            .andExpect(status().isUnauthorized());
    }
    
    @Test
    @WithMockUser(roles = "USER")
    void deletar_comUsuarioComum_deveRetornar403() throws Exception {
        mockMvc.perform(delete("/api/tarefas/1"))
            .andExpect(status().isForbidden());
    }
}
```

## Teste com JWT Real

```java
@SpringBootTest
@AutoConfigureMockMvc
class AuthControllerTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @Test
    void login_comCredenciaisValidas_deveRetornarToken() throws Exception {
        RegisterRequest register = new RegisterRequest(
            "test@test.com", "password", "Test User");
        
        mockMvc.perform(post("/api/auth/register")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(register)));
        
        LoginRequest login = new LoginRequest("test@test.com", "password");
        
        mockMvc.perform(post("/api/auth/login")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(login)))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.token").exists());
    }
}
```
