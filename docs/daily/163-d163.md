## Consumer (Frontend)

```java
@ExtendWith(PactConsumerTestExt.class)
class TarefaConsumerPactTest {
    
    @Pact(consumer = "stayhard-frontend", provider = "stayhard-api")
    V4Pact listarTarefasPact(PactDslWithProvider builder) {
        return builder
            .uponReceiving("uma requisição para listar tarefas")
            .path("/api/tarefas")
            .method("GET")
            .willRespondWith()
            .status(200)
            .headers(Map.of("Content-Type", "application/json"))
            .body(newJsonArray(o -> {
                o.object(ob -> {
                    ob.stringType("nome", "Estudar Java");
                    ob.stringType("categoria", "ESTUDO");
                    ob.stringType("prioridade", "ALTA");
                    ob.booleanType("concluida", false);
                });
            }).build())
            .toPact(V4Pact.class);
    }
    
    @Test
    void runTest(PactTestExecutionContext context) {
        new PactVerificationResult.Ok().applyTo(context, null);
    }
}
```
