## Queries customizadas

```java
public interface TarefaRepository extends JpaRepository<Tarefa, Long> {
    
    @Query("SELECT t FROM Tarefa t JOIN FETCH t.projeto WHERE t.id = :id")
    Optional<Tarefa> buscarPorIdComProjeto(@Param("id") Long id);
    
    @Query("SELECT t FROM Tarefa t JOIN FETCH t.tags WHERE t.id = :id")
    Optional<Tarefa> buscarPorIdComTags(@Param("id") Long id);
    
    List<Tarefa> findByProjetoId(Long projetoId);
    
    @Query("SELECT t FROM Tarefa t WHERE t.concluida = false ORDER BY t.prioridade DESC, t.criadoEm ASC")
    List<Tarefa> pendentesPorPrioridade();
}
```

## Service com Transaction

```java
@Service
@RequiredArgsConstructor
@Transactional
public class TarefaService {
    
    private final TarefaRepository repository;
    private final TagRepository tagRepository;
    
    public Tarefa criar(CriarTarefaRequest request) {
        Tarefa tarefa = new Tarefa();
        tarefa.setNome(request.nome());
        tarefa.setCategoria(request.categoria());
        tarefa.setPrioridade(request.prioridade());
        
        if (request.tagIds() != null) {
            List<Tag> tags = tagRepository.findAllById(request.tagIds());
            tags.forEach(tarefa::adicionarTag);
        }
        
        return repository.save(tarefa);
    }
}
```
