## Conceito do Dia

## Uso

```java
@RestController
public class TarefaController {
    
    @Autowired
    private MessageSource messageSource;
    
    @GetMapping("/api/tarefas/{id}")
    public ResponseEntity<Tarefa> buscar(
            @PathVariable Long id,
            @RequestHeader(value = "Accept-Language", defaultValue = "pt-BR") Locale locale) {
        
        Optional<Tarefa> tarefa = service.buscarPorId(id);
        
        if (tarefa.isEmpty()) {
            String msg = messageSource.getMessage(
                "erro.nao-encontrado", 
                new Object[]{"Tarefa", id}, 
                locale
            );
            throw new RecursoNaoEncontradoException(msg);
        }
        
        return ResponseEntity.ok(tarefa.get());
    }
}
```

## Frontend (React)

```bash
npm install react-i18next i18next
```

```javascript
// i18n.js
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import ptBR from './locales/pt-BR.json';
import es from './locales/es.json';

i18n
  .use(initReactI18next)
  .init({
    resources: {
      'pt-BR': { translation: ptBR },
      'es': { translation: es }
    },
    lng: 'pt-BR',
    fallbackLng: 'pt-BR',
    interpolation: {
      escapeValue: false
    }
  });

export default i18n;
```

## Componente

```jsx
import { useTranslation } from 'react-i18next';

function TarefasPage() {
  const { t, i18n } = useTranslation();
  
  return (
    <div>
      <h1>{t('tarefas.titulo')}</h1>
      
      <select 
        value={i18n.language}
        onChange={(e) => i18n.changeLanguage(e.target.value)}
      >
        <option value="pt-BR">Português</option>
        <option value="es">Español</option>
      </select>
      
      {tarefas.length === 0 ? (
        <p>{t('tarefas.empty')}</p>
      ) : (
        <ul>
          {tarefas.map(t => (
            <li key={t.id}>{t.nome}</li>
          ))}
        </ul>
      )}
    </div>
  );
}
```
