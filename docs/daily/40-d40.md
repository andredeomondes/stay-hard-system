### 📁 Exercício: Repository no Stay Hard

// 1. Interface Repository
interface UsuarioRepository {
    Optional<Usuario> findById(Long id);
    Optional<Usuario> findByEmail(String email);
    List<Usuario> findAll();
    void save(Usuario usuario);
    void delete(Long id);
}

// 2. Implementação CSV

}

// 3. Implementação JDBC
class JdbcUsuarioRepository implements UsuarioRepository {
    private DataSource dataSource;

        String sql = "SELECT * FROM usuarios WHERE email = ?";
        // Query JDBC
        return Optional.empty();
    }
    
    @Override
    public void save(Usuario u) {
        // INSERT JDBC
    }
}

// 4. Service usa Repository (não sabe a implementação)
class UsuarioService {
    private UsuarioRepository repository;
    
    public UsuarioService(UsuarioRepository repository) {
        this.repository = repository;
    }
    
    public Optional<Usuario> buscarPorEmail(String email) {
        return repository.findByEmail(email);
    }
}

// Uso
UsuarioRepository repo = new JdbcUsuarioRepository(); // Troca fácil!
UsuarioService service = new UsuarioService(repo);
