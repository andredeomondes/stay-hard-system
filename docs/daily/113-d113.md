# Dia 113 — React: useState e useEffect

## Conceito do Dia

**useState** gerencia estado local de componentes. **useEffect** executa efeitos colaterais (fetch, timers, subscriptions).

## useState

```jsx
function Contador() {
    const [count, setCount] = useState(0);
    
    return (
        <div>
            <p>Contagem: {count}</p>
            <button onClick={() => setCount(count + 1)}>+</button>
            <button onClick={() => setCount(c => c - 1)}>-</button>
            <button onClick={() => setCount(0)}>Reset</button>
        </div>
    );
}
```

## useEffect

```jsx
function Tarefas() {
    const [tarefas, setTarefas] = useState([]);
    const [loading, setLoading] = useState(true);
    const [erro, setErro] = useState(null);
    
    useEffect(() => {
        async function fetchTarefas() {
            try {
                const data = await api.listarTarefas();
                setTarefas(data);
            } catch (e) {
                setErro(e.message);
            } finally {
                setLoading(false);
            }
        }
        
        fetchTarefas();
    }, []); // array vazio = executa só uma vez (mount)
    
    if (loading) return <p>Carregando...</p>;
    if (erro) return <p>Erro: {erro}</p>;
    
    return (
        <ul>
            {tarefas.map(t => <li key={t.id}>{t.nome}</li>)}
        </ul>
    );
}
```

## Cleanup

```jsx
function EffectCleanup() {
    const [width, setWidth] = useState(window.innerWidth);
    
    useEffect(() => {
        function handleResize() {
            setWidth(window.innerWidth);
        }
        
        window.addEventListener('resize', handleResize);
        
        // Cleanup: remove listener quando componente desmontar
        return () => window.removeEventListener('resize', handleResize);
    }, []);
    
    return <p>Largura da janela: {width}px</p>;
}
```

## Dependências
