## Repository

```kotlin
// Java
public interface TarefaRepository extends JpaRepository<Tarefa, Long> {
    List<Tarefa> findByCategoria(Categoria categoria);
}

// Kotlin
interface TarefaRepository : JpaRepository<Tarefa, Long> {
    fun findByCategoria(categoria: Categoria): List<Tarefa>
    fun findByNomeContainingIgnoreCase(termo: String): List<Tarefa>
}
```

## Service

```kotlin
// Kotlin com Spring
@Service
@Transactional
class TarefaService(
    private val repository: TarefaRepository
) {
    fun criar(request: CriarTarefaRequest): Tarefa {
        val tarefa = Tarefa(
            nome = request.nome,
            categoria = request.categoria,
            prioridade = request.prioridade,
            criadoEm = LocalDateTime.now()
        )
        return repository.save(tarefa)
    }
    
    fun listarPorCategoria(categoria: Categoria) = 
        repository.findByCategoria(categoria)
    
    fun deletar(id: Long) {
        if (!repository.existsById(id)) {
            throw RecursoNaoEncontradoException("Tarefa", id)
        }
        repository.deleteById(id)
    }
}
```

## Controller

```kotlin
@RestController
@RequestMapping("/api/tarefas")
class TarefaController(
    private val service: TarefaService
) {
    @GetMapping
    fun listar(
        @RequestParam(required = false) categoria: Categoria?
    ) = if (categoria != null) {
        service.listarPorCategoria(categoria)
    } else {
        service.listarTodas()
    }
    
    @GetMapping("/{id}")
    fun buscarPorId(@PathVariable id: Long) = 
        service.buscarPorId(id)
            ?: throw RecursoNaoEncontradoException("Tarefa", id)
    
    @PostMapping
    @ResponseStatus(CREATED)
    fun criar(@RequestBody @Valid request: CriarTarefaRequest) = 
        service.criar(request)
    
    @DeleteMapping("/{id}")
    @ResponseStatus(NO_CONTENT)
    fun deletar(@PathVariable id: Long) = service.deletar(id)
}
```
