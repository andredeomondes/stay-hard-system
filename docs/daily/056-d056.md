# Dia 56 — MVC: Model-View-Controller

## Conceito do Dia

**MVC** é um padrão arquitetural que separa:
- **Model**: dados e lógica de negócio
- **View**: apresentação (JSP, HTML)
- **Controller**: recebe requisições e coordena

## Controller

```java
@WebServlet("/tarefas/*")
public class TarefaController extends HttpServlet {
    private TarefaService service = new TarefaService();
    
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        
        String action = req.getPathInfo();
        
        switch (action) {
            case "/nova" -> mostrarFormulario(req, resp);
            case "/" -> listar(req, resp);
            default -> resp.sendRedirect("/tarefas");
        }
    }
    
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        
        String nome = req.getParameter("nome");
        String catStr = req.getParameter("categoria");
        Categoria cat = Categoria.valueOf(catStr.toUpperCase());
        
        service.criarTarefa(nome, cat);
        resp.sendRedirect("/tarefas");
    }
    
    private void listar(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        List<Tarefa> tarefas = service.listarTodas();
        req.setAttribute("tarefas", tarefas);
        req.getRequestDispatcher("/WEB-INF/views/lista.jsp").forward(req, resp);
    }
    
    private void mostrarFormulario(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        req.getRequestDispatcher("/WEB-INF/views/form.jsp").forward(req, resp);
    }
}
```
