class EmailValidatorTest {
    
    private EmailValidator validator;
    
    @BeforeEach
    void setup() {
        validator = new EmailValidator();
    }
    
    // ✅ VÁLIDOS
    @Test
    @DisplayName("Email válido simples")

    }

    @DisplayName("Email com subdomain")
    void testEmailComSubdomain() {
        assertTrue(validator.isValid("user@mail.dominio.com"));
    }

    @DisplayName("Email com ponto no username")
    void testEmailComPonto() {
        assertTrue(validator.isValid("nome.sobrenome@dominio.com"));
    }
    
    @Test
    @DisplayName("Email com mais ponto")
    void testEmailMaisPonto() {
        assertTrue(validator.isValid("usuario@empresa.dominio.com.br"));

    // ❌ INVÁLIDOS

        assertFalse(validator.isValid("usuariodominio.com"));

    void testEmailSemUsername() {
        assertFalse(validator.isValid("@dominio.com"));
    }
    
    @Test
    @DisplayName("Email sem domínio")
    void testEmailSemDominio() {
        assertFalse(validator.isValid("usuario@"));

### 3. Implemente para passar os testes

```java
class EmailValidator {
    
    public boolean isValid(String email) {
        if (email == null || email.isBlank()) {
            return false;
        }
        
        // Regex básico para email
        String regex = "^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$";
        return email.matches(regex);
    }
}
```
