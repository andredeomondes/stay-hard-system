### 2. CpfValidator.java

**Regras de validação de CPF:**
```
CPF válido: 123.456.789-00

Dígitos verificadores:
1. Multiplicar cada dígito por pesos decrescentes (11,10,9...)
2. Somar resultados
3. Pegar resto da divisão por 11
4. Primeiro dígito = 11 - resto
5. Repetir para segundo dígito
```

```java
package com.estudo;

import java.util.InputMismatchException;

public class CpfValidator {
    
    public boolean isValid(String cpf) {
        if (cpf == null) {
            return false;
        }
        
        // Remove formatação
        String cpfNumerico = cpf.replaceAll("[^0-9]", "");
        
        // Deve ter 11 dígitos
        if (cpfNumerico.length() != 11) {
            return false;
        }
        
        // Não pode ter todos dígitos iguais (ex: 111.111.111-11)
        if (cpfNumerico.matches("(\\d)\\1{10}")) {
            return false;
        }
        
        // Valida primeiro dígito
        if (!validarDigito(cpfNumerico, 9)) {
            return false;
        }
        
        // Valida segundo dígito
        if (!validarDigito(cpfNumerico, 10)) {
            return false;
        }
        
        return true;
    }
    
    private boolean validarDigito(String cpf, int peso) {
        int soma = 0;
        
        for (int i = 0; i < peso; i++) {
            int digito = Character.getNumericValue(cpf.charAt(i));
            soma += digito * (peso + 1 - i);
        }
        
        int resto = soma % 11;
        int digitoVerificador = resto < 2 ? 0 : 11 - resto;
        
        return digitoVerificador == Character.getNumericValue(cpf.charAt(peso));
    }
    
    public String formatar(String cpf) {
        if (cpf == null || cpf.length() != 11) {
            return cpf;
        }
        
        String nums = cpf.replaceAll("[^0-9]", "");
        return String.format("%s.%s.%s-%s",
            nums.substring(0, 3),
            nums.substring(3, 6),
            nums.substring(6, 9),
            nums.substring(9, 11)
        );
    }
}
```

### 3. CpfValidatorTest.java

```java
package com.estudo;

import org.junit.jupiter.api.*;
import static org.junit.jupiter.api.Assertions.*;

class CpfValidatorTest {
    
    private CpfValidator validator;
    
    @BeforeEach
    void setup() {
        validator = new CpfValidator();
    }
    
    // CPFs VÁLIDOS
    @Test
    @DisplayName("CPF válido: 11144477735")
    void testCpfValido() {
        assertTrue(validator.isValid("11144477735"));
    }
    
    @Test
    @DisplayName("CPF válido com formatação")
    void testCpfValidoFormatado() {
        assertTrue(validator.isValid("111.444.777-35"));
    }
    
    @Test
    @DisplayName("CPF válido: 12345678909")
    void testCpfValido2() {
        assertTrue(validator.isValid("12345678909"));
    }
    
    // CPFs INVÁLIDOS
    @Test
    @DisplayName("CPF nulo é inválido")
    void testCpfNulo() {
        assertFalse(validator.isValid(null));
    }
    
    @Test
    @DisplayName("CPF vazio é inválido")
    void testCpfVazio() {
        assertFalse(validator.isValid(""));
    }
    
    @Test
    @DisplayName("CPF com menos de 11 dígitos é inválido")
    void testCpfCurto() {
        assertFalse(validator.isValid("1234567890"));
    }
    
    @Test
    @DisplayName("CPF com mais de 11 dígitos é inválido")
    void testCpfLongo() {
        assertFalse(validator.isValid("123456789012"));
    }
    
    @Test
    @DisplayName("CPF com todos dígitos iguais é inválido")
    void testCpfTodosIguais() {
        assertFalse(validator.isValid("11111111111"));
        assertFalse(validator.isValid("00000000000"));
        assertFalse(validator.isValid("99999999999"));
    }
    
    @Test
    @DisplayName("CPF com dígitos verificadores errados")
    void testCpfDigitosErrados() {
        assertFalse(validator.isValid("12345678900")); // Último dígito errado
    }
    
    // FORMATAÇÃO
    @Test
    @DisplayName("Formatar CPF numérico")
    void testFormatar() {
        assertEquals("123.456.789-09", validator.formatar("12345678909"));
    }
    
    @Test
    @DisplayName("Formatar CPF já formatado")
    void testFormatarJaFormatado() {
        assertEquals("123.456.789-09", validator.formatar("123.456.789-09"));
    }
}
```
