## Fundamentos

```javascript
// Variáveis
let nome = "João";        // pode mudar
const PI = 3.14159;       // não pode mudar

// Tipos
let texto = "Olá";        // string
let numero = 42;          // number
let booleano = true;      // boolean
let lista = [1, 2, 3];    // array
let pessoa = {            // objeto
    nome: "João",
    idade: 25
};

// Funções
function saudar(nome) {
    return `Olá, ${nome}!`;
}

// Arrow function
const dobrar = (n) => n * 2;

// Condicionais
if (numero > 10) {
    console.log("Maior que 10");
} else if (numero === 10) {
    console.log("Igual a 10");
} else {
    console.log("Menor que 10");
}

// Ternário
const status = numero > 0 ? "positivo" : "negativo";

// Loops
for (let i = 0; i < 5; i++) {
    console.log(i);
}

lista.forEach(item => {
    console.log(item);
});

// Map
const dobrados = lista.map(n => n * 2);

// Filter
const pares = lista.filter(n => n % 2 === 0);
```

## DOM Manipulation

```javascript
// Selecionar elementos
const titulo = document.querySelector('h1');
const listaTarefas = document.querySelectorAll('.tarefa');

// Modificar conteúdo
titulo.textContent = "Stay Hard API";

// Criar elemento
const novaDiv = document.createElement('div');
novaDiv.textContent = "Nova tarefa";
novaDiv.className = "tarefa";

// Adicionar ao DOM
document.querySelector('main').appendChild(novaDiv);

// Eventos
document.querySelector('#btn-add').addEventListener('click', () => {
    console.log("Clicado!");
});
```
