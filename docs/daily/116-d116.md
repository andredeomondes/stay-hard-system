## Setup

```bash
npm install @tanstack/react-query
```

```jsx
// main.jsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

const queryClient = new QueryClient({
    defaultOptions: {
        queries: {
            staleTime: 1000 * 60 * 5, // 5 minutos
            retry: 1
        }
    }
});

ReactDOM.createRoot(document.getElementById('root')).render(
    <QueryClientProvider client={queryClient}>
        <App />
    </QueryClientProvider>
);
```

## useQuery

```jsx
import { useQuery } from '@tanstack/react-query';

function Tarefas() {
    const { data: tarefas, isLoading, error, refetch } = useQuery({
        queryKey: ['tarefas'],
        queryFn: () => api.listarTarefas()
    });
    
    if (isLoading) return <p>Carregando...</p>;
    if (error) return <p>Erro: {error.message}</p>;
    
    return (
        <div>
            <button onClick={() => refetch()}>Atualizar</button>
            {tarefas.map(t => <TarefaCard key={t.id} tarefa={t} />)}
        </div>
    );
}
```

## useMutation

```jsx
import { useMutation, useQueryClient } from '@tanstack/react-query';

function NovaTarefaForm() {
    const queryClient = useQueryClient();
    
    const criarMutation = useMutation({
        mutationFn: (dados) => api.criarTarefa(dados),
        onSuccess: () => {
            // Invalidar e refetch
            queryClient.invalidateQueries({ queryKey: ['tarefas'] });
            // ou
            queryClient.setQueryData(['tarefas'], (old) => [...old, newData]);
        },
        onError: (error) => {
            alert(error.message);
        }
    });
    
    const handleSubmit = (dados) => {
        criarMutation.mutate(dados);
    };
    
    return (
        <form onSubmit={handleSubmit}>
            <input {...register('nome')} />
            <button type="submit" disabled={criarMutation.isPending}>
                {criarMutation.isPending ? 'Salvando...' : 'Criar'}
            </button>
        </form>
    );
}
```
