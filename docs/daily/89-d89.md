## Teste de Controller

```java
@WebMvcTest(TarefaController.class)
@AutoConfigureMockMvc
class TarefaControllerTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @MockBean
    private TarefaService service;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @Test
    void listar_deveRetornar200() throws Exception {
        List<Tarefa> tarefas = List.of(
            criarTarefa(1L, "Estudar"),
            criarTarefa(2L, "Trabalhar")
        );
        
        when(service.listarTodas()).thenReturn(tarefas);
        
        mockMvc.perform(get("/api/tarefas")
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.length()").value(2))
            .andExpect(jsonPath("$[0].nome").value("Estudar"))
            .andDo(print());
    }
    
    @Test
    void criar_comDadosValidos_deveRetornar201() throws Exception {
        CriarTarefaRequest request = new CriarTarefaRequest(
            "Nova Tarefa", Categoria.ESTUDO, Prioridade.ALTA);
        
        Tarefa tarefaSalva = criarTarefa(1L, "Nova Tarefa");
        
        when(service.criar(any())).thenReturn(tarefaSalva);
        
        mockMvc.perform(post("/api/tarefas")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(request)))
            .andExpect(status().isCreated())
            .andExpect(header().exists("Location"));
    }
    
    @Test
    void criar_comNomeVazio_deveRetornar400() throws Exception {
        CriarTarefaRequest request = new CriarTarefaRequest(
            "", Categoria.ESTUDO, Prioridade.ALTA);
        
        mockMvc.perform(post("/api/tarefas")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(request)))
            .andExpect(status().isBadRequest())
            .andExpect(jsonPath("$.detalhes[0]").value("nome: Nome é obrigatório"));
    }
    
    @Test
    void buscarPorId_naoEncontrado_deveRetornar404() throws Exception {
        when(service.buscarPorId(999L)).thenReturn(Optional.empty());
        
        mockMvc.perform(get("/api/tarefas/999"))
            .andExpect(status().isNotFound());
    }
}
```
