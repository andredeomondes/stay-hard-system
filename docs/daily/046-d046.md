# Dia 46 — Observer Pattern

## Conceito do Dia

Quando um objeto muda de estado, todos os seus "observadores" são notificados automaticamente. É a base do padrão **pub/sub**.

## Código: Sistema de Tarefas com Notificações

```java
import java.util.ArrayList;
import java.util.List;

// Observer interface
interface Observador {
    void atualizar(String mensagem);
}

// Subject (Observable)
class TarefaService {
    private List<Observador> observadores = new ArrayList<>();
    
    void adicionarObservador(Observador obs) {
        observadores.add(obs);
    }
    
    void removerObservador(Observador obs) {
        observadores.remove(obs);
    }
    
    void criarTarefa(String nome) {
        // ... lógica de criar tarefa
        notificarObservadores("Tarefa criada: " + nome);
    }
    
    void concluirTarefa(Long id) {
        // ... lógica de concluir
        notificarObservadores("Tarefa concluída! ID: " + id);
    }
    
    private void notificarObservadores(String msg) {
        observadores.forEach(obs -> obs.atualizar(msg));
    }
}

// Observadores concretos
class NotificadorConsole implements Observador {
    @Override
    public void atualizar(String msg) {
        System.out.println("[CONSOLE] " + msg);
    }
}

class NotificadorEmail implements Observador {
    private String email;
    
    NotificadorEmail(String email) { this.email = email; }
    
    @Override
    public void atualizar(String msg) {
        System.out.println("[EMAIL para " + email + "] " + msg);
    }
}

// Uso
public class Main {
    public static void main(String[] args) {
        TarefaService service = new TarefaService();
        
        service.adicionarObservador(new NotificadorConsole());
        service.adicionarObservador(new NotificadorEmail("user@email.com"));
        
        service.criarTarefa("Estudar Java");
        service.concluirTarefa(1L);
    }
}
```
