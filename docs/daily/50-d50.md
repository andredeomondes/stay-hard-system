# Dia 50 — DDD: Services e Repositories

## Conceito do Dia

**Service**: Orquestra operações de negócio, não tem estado próprio.
**Repository**: Abstração de persistência. Parece uma coleção em memória, mas persiste.

## Código: Repository e Service

```java
// INTERFACE DO REPOSITORY
public interface TarefaRepository {
    Tarefa salvar(Tarefa tarefa);
    Optional<Tarefa> buscarPorId(Long id);
    List<Tarefa> buscarTodos();
    void deletar(Long id);
    List<Tarefa> buscarPorCategoria(Categoria cat);
}

// IMPLEMENTAÇÃO EM MEMÓRIA
public class TarefaRepositoryImpl implements TarefaRepository {
    private final List<Tarefa> tarefas = new ArrayList<>();
    private Long idCounter = 1L;
    
    @Override
    public Tarefa salvar(Tarefa tarefa) {
        if (tarefa.getId() == null) {
            tarefa.setId(idCounter++);
            tarefas.add(tarefa);
        } else {
            tarefas.removeIf(t -> t.getId().equals(tarefa.getId()));
            tarefas.add(tarefa);
        }
        return tarefa;
    }
    
    @Override
    public Optional<Tarefa> buscarPorId(Long id) {
        return tarefas.stream()
            .filter(t -> t.getId().equals(id))
            .findFirst();
    }
    
    @Override
    public List<Tarefa> buscarTodos() {
        return new ArrayList<>(tarefas);
    }
    
    @Override
    public void deletar(Long id) {
        tarefas.removeIf(t -> t.getId().equals(id));
    }
    
    @Override
    public List<Tarefa> buscarPorCategoria(Categoria cat) {
        return tarefas.stream()
            .filter(t -> t.getCategoria() == cat)
            .toList();
    }
}

// SERVICE - orquestra, não persiste
public class TarefaService {
    private final TarefaRepository repository;
    
    public TarefaService(TarefaRepository repository) {
        this.repository = repository;
    }
    
    public Tarefa criarTarefa(String nome, Categoria cat, Prioridade p) {
        Tarefa tarefa = Tarefa.builder()
            .nome(nome)
            .categoria(cat)
            .prioridade(p)
            .criadoEm(LocalDateTime.now())
            .concluida(false)
            .build();
        
        return repository.salvar(tarefa);
    }
    
    public void concluirTarefa(Long id) {
        Tarefa tarefa = repository.buscarPorId(id)
            .orElseThrow(() -> new TarefaNaoEncontradaException(id));
        
        tarefa.setConcluida(true);
        repository.salvar(tarefa);
    }
}
```
