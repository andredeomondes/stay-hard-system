## Código: TarefaServlet

```java
@WebServlet("/api/tarefas/*")
public class TarefaServlet extends HttpServlet {
    private TarefaService service = new TarefaService();
    
    // GET /api/tarefas
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        
        String pathInfo = req.getPathInfo();
        
        if (pathInfo == null || pathInfo.equals("/")) {
            // Listar todas
            List<Tarefa> tarefas = service.listarTodas();
            enviarJson(resp, tarefas);
        } else {
            // Buscar por ID
            Long id = Long.parseLong(pathInfo.substring(1));
            Tarefa tarefa = service.buscarPorId(id);
            enviarJson(resp, tarefa);
        }
    }
    
    // POST /api/tarefas
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) 
            throws ServletException, IOException {
        
        // Ler body JSON
        BufferedReader reader = req.getReader();
        String body = reader.lines().reduce("", (a, b) -> a + b);
        
        // Parsear (simples, sem Jackson por enquanto)
        String nome = extrairJson(body, "nome");
        String catStr = extrairJson(body, "categoria");
        
        Categoria cat = Categoria.valueOf(catStr.toUpperCase());
        Tarefa nova = service.criarTarefa(nome, cat);
        
        resp.setStatus(201);
        enviarJson(resp, nova);
    }
    
    private void enviarJson(HttpServletResponse resp, Object obj) 
            throws IOException {
        resp.setContentType("application/json");
        resp.setCharacterEncoding("UTF-8");
        resp.getWriter().write(toJson(obj));
    }
    
    private String toJson(Object obj) {
        // Simulando conversão - usar Jackson na prática
        return "{\"id\": 1, \"nome\": \"teste\"}";
    }
    
    private String extrairJson(String json, String campo) {
        // Parsing simples - na prática use Jackson
        String regex = "\"" + campo + "\"\\s*:\\s*\"([^\"]+)\"";
        var m = Pattern.compile(regex).matcher(json);
        return m.find() ? m.group(1) : null;
    }
}
```
