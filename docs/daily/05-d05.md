### 2. ChatUtils.java — Utilitários

- [ ] Classe com métodos auxiliares

```java
package com.estudo;

import java.io.*;
import java.net.*;

public class ChatUtils {
    
    public static PrintWriter getWriter(Socket socket) {
        return new PrintWriter(socket.getOutputStream(), true);
    }
    
    public static BufferedReader getReader(Socket socket) {
        return new BufferedReader(new InputStreamReader(socket.getInputStream()));
    }
    
    public static String readLine(BufferedReader reader) throws IOException {
        String line = reader.readLine();
        if (line == null) {
            throw new IOException("Conexão encerrada");
        }
        return line;
    }
    
    public static void print(String message) {
        System.out.println(message);
    }
    
    public static void printServer(String message) {
        System.out.println("[SERVER] " + message);
    }
    
    public static void printClient(String message) {
        System.out.println("[CLIENT] " + message);
    }
}
```

### 3. ChatServer.java — Servidor

- [ ] Criar ServerSocket na porta 8080
- [ ] Aguardar conexão
- [ ] Loop de chat (enviar/receber)
- [ ] Tratar encerramento

```java
package com.estudo;

import java.io.*;
import java.net.*;

public class ChatServer {
    
    private static final int PORTA = 8080;
    
    public static void main(String[] args) {
        try (ServerSocket serverSocket = new ServerSocket(PORTA)) {
            ChatUtils.print("Servidor de chat iniciado na porta " + PORTA);
            ChatUtils.print("Aguardando conexão...");
            
            try (Socket clientSocket = serverSocket.accept();
                 PrintWriter out = ChatUtils.getWriter(clientSocket);
                 BufferedReader in = ChatUtils.getReader(clientSocket)) {
                
                ChatUtils.printServer("Cliente conectado!");
                out.println("=== BEM-VINDO AO CHAT ===");
                out.println("Digite 'sair' para encerrar");
                
                String mensagem;
                while ((mensagem = ChatUtils.readLine(in)) != null) {
                    ChatUtils.printServer("Cliente: " + mensagem);
                    
                    if (mensagem.equalsIgnoreCase("sair")) {
                        out.println("Você saiu do chat. Até mais!");
                        break;
                    }
                    
                    // Eco com resposta
                    out.println("Servidor recebeu: " + mensagem);
                }
                
                ChatUtils.printServer("Conexão encerrada.");
            }
            
        } catch (IOException e) {
            ChatUtils.print("Erro no servidor: " + e.getMessage());
        }
    }
}
```

### 4. ChatClient.java — Cliente

- [ ] Conectar ao servidor (localhost:8080)
- [ ] Loop de chat (ler do console, enviar)
- [ ] Receber e mostrar mensagens
- [ ] Tratar encerramento

```java
package com.estudo;

import java.io.*;
import java.net.*;

public class ChatClient {
    
    private static final String HOST = "localhost";
    private static final int PORTA = 8080;
    
    public static void main(String[] args) {
        try (Socket socket = new Socket(HOST, PORTA);
             PrintWriter out = ChatUtils.getWriter(socket);
             BufferedReader in = ChatUtils.getReader(socket);
             BufferedReader console = new BufferedReader(
                 new InputStreamReader(System.in))) {
            
            ChatUtils.printClient("Conectado ao servidor!");
            
            // Thread para receber mensagens
            Thread readerThread = new Thread(() -> {
                try {
                    String msgServidor;
                    while ((msgServidor = in.readLine()) != null) {
                        ChatUtils.printServer(msgServidor);
                        if (msgServidor.contains("saiu")) break;
                    }
                } catch (IOException e) {
                    // Conexão fechada
                }
            });
            readerThread.start();
            
            // Loop para enviar mensagens
            String mensagem;
            while ((mensagem = console.readLine()) != null) {
                out.println(mensagem);
                if (mensagem.equalsIgnoreCase("sair")) {
                    break;
                }
            }
            
        } catch (ConnectException e) {
            ChatUtils.printClient("Não foi possível conectar ao servidor.");
            ChatUtils.printClient("Verifique se o servidor está rodando.");
        } catch (IOException e) {
            ChatUtils.printClient("Erro: " + e.getMessage());
        }
        
        ChatUtils.printClient("Você saiu do chat.");
    }
}
```
