## Store Simples

```javascript
class Store {
    constructor() {
        this.state = {
            tarefas: [],
            usuario: null,
            token: localStorage.getItem('token'),
            loading: false,
            erro: null
        };
        this.listeners = [];
    }
    
    getState() {
        return this.state;
    }
    
    setState(newState) {
        this.state = { ...this.state, ...newState };
        this.notify();
    }
    
    subscribe(listener) {
        this.listeners.push(listener);
        return () => {
            this.listeners = this.listeners.filter(l => l !== listener);
        };
    }
    
    notify() {
        this.listeners.forEach(listener => listener(this.state));
    }
}

const store = new Store();
```

## Actions

```javascript
const actions = {
    async carregarTarefas() {
        store.setState({ loading: true, erro: null });
        try {
            const tarefas = await API.listarTarefas();
            store.setState({ tarefas, loading: false });
        } catch (e) {
            store.setState({ erro: e.message, loading: false });
        }
    },
    
    async criarTarefa(dados) {
        store.setState({ loading: true });
        try {
            await API.criarTarefa(dados);
            await this.carregarTarefas();
        } catch (e) {
            store.setState({ erro: e.message, loading: false });
        }
    },
    
    login(token) {
        localStorage.setItem('token', token);
        store.setState({ token });
    },
    
    logout() {
        localStorage.removeItem('token');
        store.setState({ token: null, tarefas: [], usuario: null });
    }
};
```

## Render baseado no State

```javascript
function render(state) {
    const { tarefas, loading, erro, token } = state;
    
    if (loading) {
        document.getElementById('app').innerHTML = '<p>Carregando...</p>';
        return;
    }
    
    if (erro) {
        document.getElementById('app').innerHTML = `<p class="erro">${erro}</p>`;
        return;
    }
    
    if (!token) {
        renderLogin();
        return;
    }
    
    renderTarefas(tarefas);
}

// Subscribe to state changes
store.subscribe(render);
```
