## useLocalStorage

```jsx
function useLocalStorage(key, initialValue) {
    const [storedValue, setStoredValue] = useState(() => {
        try {
            const item = window.localStorage.getItem(key);
            return item ? JSON.parse(item) : initialValue;
        } catch (error) {
            console.error(error);
            return initialValue;
        }
    });
    
    const setValue = (value) => {
        try {
            const valueToStore = value instanceof Function ? value(storedValue) : value;
            setStoredValue(valueToStore);
            window.localStorage.setItem(key, JSON.stringify(valueToStore));
        } catch (error) {
            console.error(error);
        }
    };
    
    return [storedValue, setValue];
}

// Uso
function App() {
    const [tarefas, setTarefas] = useLocalStorage('tarefas', []);
}
```

## useDebounce

```jsx
function useDebounce(value, delay = 500) {
    const [debouncedValue, setDebouncedValue] = useState(value);
    
    useEffect(() => {
        const handler = setTimeout(() => {
            setDebouncedValue(value);
        }, delay);
        
        return () => clearTimeout(handler);
    }, [value, delay]);
    
    return debouncedValue;
}

// Uso
function Busca() {
    const [termo, setTermo] = useState('');
    const debouncedTermo = useDebounce(termo, 300);
    
    useEffect(() => {
        if (debouncedTermo) {
            search(debouncedTermo);
        }
    }, [debouncedTermo]);
}
```

## useTarefas

```jsx
function useTarefas() {
    const queryClient = useQueryClient();
    
    const tarefasQuery = useQuery({
        queryKey: ['tarefas'],
        queryFn: () => api.listarTarefas()
    });
    
    const criarMutation = useMutation({
        mutationFn: (dados) => api.criarTarefa(dados),
        onSuccess: () => {
            queryClient.invalidateQueries({ queryKey: ['tarefas'] });
        }
    });
    
    const concluirMutation = useMutation({
        mutationFn: (id) => api.concluirTarefa(id),
        onSuccess: () => {
            queryClient.invalidateQueries({ queryKey: ['tarefas'] });
        }
    });
    
    const deletarMutation = useMutation({
        mutationFn: (id) => api.deletarTarefa(id),
        onSuccess: () => {
            queryClient.invalidateQueries({ queryKey: ['tarefas'] });
        }
    });
    
    return {
        tarefas: tarefasQuery.data ?? [],
        isLoading: tarefasQuery.isLoading,
        criar: criarMutation.mutate,
        concluir: concluirMutation.mutate,
        deletar: deletarMutation.mutate,
        isCriando: criarMutation.isPending
    };
}

// Uso
function TarefasPage() {
    const { tarefas, isLoading, criar, concluir, deletar } = useTarefas();
    // ...
}
```
