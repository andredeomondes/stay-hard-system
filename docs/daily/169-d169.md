## OpenAPI 3.0

```yaml
openapi: 3.0.3
info:
  title: Stay Hard API
  description: API para gerenciador de tarefas
  version: 1.0.0
  contact:
    email: api@stayhard.com

servers:
  - url: https://api.stayhard.com/v1
    description: Produção
  - url: http://localhost:8080/api/v1
    description: Desenvolvimento

paths:
  /tarefas:
    get:
      summary: Listar tarefas
      description: Retorna lista paginada de tarefas
      tags: [Tarefas]
      parameters:
        - name: categoria
          in: query
          schema:
            type: string
            enum: [ESTUDO, TRABALHO, SAUDE, LAZER]
        - $ref: '#/components/parameters/Page'
        - $ref: '#/components/parameters/Size'
      responses:
        '200':
          description: Lista de tarefas
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/TarefaPage'
        '401':
          $ref: '#/components/responses/Unauthorized'
    
    post:
      summary: Criar tarefa
      tags: [Tarefas]
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/CriarTarefaRequest'
      responses:
        '201':
          description: Tarefa criada
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Tarefa'

components:
  schemas:
    Tarefa:
      type: object
      properties:
        id:
          type: integer
          format: int64
        nome:
          type: string
          maxLength: 100
        categoria:
          $ref: '#/components/schemas/Categoria'
        prioridade:
          $ref: '#/components/schemas/Prioridade'
        concluida:
          type: boolean
        criadoEm:
          type: string
          format: date-time

    Categoria:
      type: string
      enum: [ESTUDO, TRABALHO, SAUDE, LAZER]

    CriarTarefaRequest:
      type: object
      required:
        - nome
        - categoria
        - prioridade
      properties:
        nome:
          type: string
        categoria:
          $ref: '#/components/schemas/Categoria'
        prioridade:
          $ref: '#/components/schemas/Prioridade'
```
