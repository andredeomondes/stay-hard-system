## Tarefa.java

```java
@Entity
@Table(name = "tarefas")
public class Tarefa {
    
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(nullable = false, length = 100)
    private String nome;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private Categoria categoria;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private Prioridade prioridade;
    
    private boolean concluida;
    
    private LocalDateTime criadoEm;
    private LocalDateTime concluidoEm;
}
```

## TarefaRepository.java

@Repository
public interface TarefaRepository extends JpaRepository<Tarefa, Long> {
    
    List<Tarefa> findByCategoria(Categoria categoria);
    
    List<Tarefa> findByConcluidaFalse();
    
    List<Tarefa> findByNomeContainingIgnoreCase(String termo);
}

## TarefaService.java

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class TarefaService {
    
    private final TarefaRepository repository;
    
    public List<Tarefa> listarTodas() {
        return repository.findAll();
    }
    
    public Optional<Tarefa> buscarPorId(Long id) {
        return repository.findById(id);
    }
    
    public Tarefa criar(CriarTarefaRequest request) {
        Tarefa tarefa = new Tarefa();
        tarefa.setNome(request.nome());
        tarefa.setCategoria(request.categoria());
        tarefa.setPrioridade(request.prioridade());
        tarefa.setCriadoEm(LocalDateTime.now());
        
        return repository.save(tarefa);
    }
    
    public void deletar(Long id) {
        if (!repository.existsById(id)) {
            throw new RecursoNaoEncontradoException("tarefa", id);
        }
        repository.deleteById(id);
    }
}
```
